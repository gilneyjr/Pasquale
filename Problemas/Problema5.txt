PROCEDIMENTO mdc RECEBE (INTEIRO n, INTEIRO m, PONTEIRO INTEIRO r)
    SE m >= n ENTAO
        SE m MOD n = 0 ENTAO
            VALOR(r) := n;
        SENAO
            mdc(n, (m MOD n), r);
        FIMSE
    SENAO
        SE n MOD m = 0 ENTAO
            VALOR(r) := m;
        SENAO
            mdc(m, (n MOD m), r);
        FIMSE
    FIMSE
FIMPROCEDIMENTO

PRINCIPAL
    INTEIRO n, m;
    ESCREVA("Digite dois números positivos:\n");
    LEIA(n, m);
    SE n <= 0 OU m <= 0 ENTAO
        ESCREVA("Por favor forneça valores estritamente positivos.\n");
    SENAO
        PONTEIRO INTEIRO r;
        r := NOVO INTEIRO;
        mdc(n,m,r);
        ESCREVA(VALOR(r));
        ESCREVA("\n");
    FIMSE
FIMPRINCIPAL
