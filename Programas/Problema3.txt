//Problema 3: Crie um programa que leia duas matrizes numéricas e, quando possível, imprima a soma e o produto dessas matrizes. Caso uma operação não possa ser realizada para as matrizes lidas, imprima uma mensagem informando da impossibilidade.

FUNCAO lerMatriz RECEBE (inteiro a[lA,lA]) RET

PRINCIPAL
	INTEIRO lA, cA, lB,cB;
	LEIA(lA);
	LEIA(cA);
	LEIA(lB);
	LEIA(cB);
	INTEIRO a[lA,cA];
	INTEIRO b[lB,cB];
	INTEIRO i := 1;
	INTEIRO j := 1;
	ENQUANTO  (i <= lA) EXECUTE
		ENQUANTO (j <= cA) EXECUTE
			LEIA(a[i,j]);
			j++;
		FIMENQUANTO
		i++;
	FIMENQUANTO

	i := 1;
	j := 1;
	ENQUANTO  (i <= lB) EXECUTE
		ENQUANTO (j <= cB) EXECUTE
			LEIA(b[i,j]);
			j++;
		FIMENQUANTO
		i++;
	FIMENQUANTO

	SE (lA = lB E cA = cB) ENTAO
		i:= 1; j := i;
		ENQUANTO (i <= lA) EXECUTE
			ENQUANTO (j <= cA) EXECUTE
				ESCREVA ((TEXTO) (a[i,j] + b[i,j])  + " ");
				j++;
			FIMENQUANTO
			ESCREVA ("\n");
			i++;
		FIMENQUANTO
	SENAO
		ESCREVA("Não é possível somar essas matrizes.\n");
	FIMSE

	SE (lA = cB ) ENTAO
		i:= 1; j := i;
		INTEIRO k := 1;
		INTEIRO resposta[lA, cB];
		ENQUANTO (i <= lA) EXECUTE
			ENQUANTO (j <= cB) EXECUTE
				ENQUANTO (k <= lA) EXECUTE
					resposta[i,j] = resposta[i,j] + a[i,k] * b[k,j];
					k++;
				FIMENQUANTO
				ESCREVA ((TEXTO) resposta[i,j]  + " ");
				j++;
			FIMENQUANTO
			ESCREVA ("\n");
			i++;
		FIMENQUANTO
	SENAO
		ESCREVA("Não é possível multiplicar essas matrizes.\n");
	FIMSE
FIMPRINCIPAL
